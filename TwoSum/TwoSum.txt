public class Solution {
    public int[] twoSum(int[] nums, int target) {
     int[] B = new int[2];
    Map<Integer, Integer> map = new HashMap<Integer, Integer>();
    for (int i = 0; i < nums.length; i++) {
        if (map.containsKey(target - nums[i])) {
   
           B[1] = i;
            B[0] = map.get(target - nums[i]);
            return B;
        }
        map.put(nums[i], i );
      }
    return B;
    }
}
